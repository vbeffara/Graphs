\section{Contractions and minors}

\begin{definition}
  \label{def:minor} \lean{IsMinor} \leanok
  A graph $H$ is a minor of $G$ if it can be realized as a subgraph of a
  contraction minor of $G$, or equivalently as a contraction subgraph of a
  subgraph of $G$ (in Diestel's parlance, if $G$ contains an $IH$ as a
  subgraph).
\end{definition}
