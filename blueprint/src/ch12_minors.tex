\section{12.1 Well-quasi-ordering}

\begin{definition}[Diestel \S12.1]
  \label{def:wqo}
  A quasi-ordering is a \emph{well-quasi-ordering} (WQO) if there is no
  infinite antichain and no infinite strictly descending sequence.
\end{definition}
